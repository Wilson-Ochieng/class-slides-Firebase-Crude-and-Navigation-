📱 Step 3: Flutter App Configuration

API Configuration

dart 
// lib/config/api_config.dart
class ApiConfig {
  // Update with your ngrok URL
  static const String ngrokUrl = 'https://YOUR_NGROK_URL.ngrok-free.app';
  static const String projectId = 'demprojo';
  
  static String get baseUrl => '$ngrokUrl/$projectId/us-central1/api';
  
  static String get health => '$baseUrl/health';
  static String get initiateStkPush => '$baseUrl/initiate-stk-push';
  static String get checkPayment => '$baseUrl/check-payment';
  static String get mpesaCallback => '$baseUrl/mpesa-callback';
  
  static const bool useNgrok = true;
}

M-Pesa Service

dart 
// lib/services/mpesa_service.dart
class MpesaService {
  Future<Map<String, dynamic>> initiateStkPush({
    required String phone,
    required int amount,
    required String orderId,
  }) async {
    final url = Uri.parse(ApiConfig.initiateStkPush);
    
    final response = await http.post(
      url,
      headers: {'Content-Type': 'application/json'},
      body: jsonEncode({
        'phone': phone,
        'amount': amount,
        'orderId': orderId,
      }),
    );
    
    if (response.statusCode == 200) {
      return {'success': true, 'data': jsonDecode(response.body)};
    }
    return {'success': false, 'error': 'HTTP ${response.statusCode}'};
  }
}

🔧 Troubleshooting

❌ Port Already in Use

sudo fuser -k 4000/tcp 4400/tcp 4500/tcp 5001/tcp
pkill -f firebase
pkill -f ngrok

❌ ngrok URL Changed

# Get new URL
curl -s http://localhost:4040/api/tunnels | python3 -c "import sys, json; print(json.load(sys.stdin)['tunnels'][0]['public_url'])"

# Update ApiConfig.ngrokUrl and .env file

❌ Callback Not Received

# Check ngrok traffic
open http://localhost:4040

# Watch Firebase logs
firebase functions:log --only api --tail

❌ Network Error in Flutter

  • ✓ Ensure ngrok is running
  • ✓ Update ApiConfig.ngrokUrl
  • ✓ Restart Flutter app
  • ✓ Check internet connection
