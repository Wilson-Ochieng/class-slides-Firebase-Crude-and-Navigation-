Provider Setup — The Manager

// product_provider.dart
class ProductProvider with ChangeNotifier {
  final CollectionReference _productsCollection =
      FirebaseFirestore.instance.collection('products');

  List<ProductModel> _products = [];
  bool _isLoading = false;

  // Getters
  List<ProductModel> get products => _products;
  bool get isLoading => _isLoading;
}

Key Concepts

  • ChangeNotifier — Notifies the UI when data changes
  • CollectionReference — Points to a Firestore collection
  • _products — Local in-memory cache of product list
  • notifyListeners() — Triggers UI rebuild on state change

READ Operation

// Fetching all products from Firestore
Future<void> fetchProducts() async {
  _setLoading(true);
  try {
    // Get all documents from collection
    QuerySnapshot snapshot = await _productsCollection.get();

    // Convert each doc to a ProductModel
    _products = snapshot.docs.map((doc) {
      return ProductModel.fromMap(
        doc.data() as Map<String, dynamic>, doc.id);
    }).toList();

    notifyListeners(); // Trigger UI rebuild
  } finally { _setLoading(false); }
}

💡 Pro Tip: Use Consumer<ProductProvider> to rebuild only the widget subtree that depends on the provider, not the whole screen.

Slide 06 / 15

UPDATE Operation

// Modifying an existing Firestore document
Future<String?> updateProduct({
  required String productId,
  required String name,
  required double price,
  File? newImageFile,
  required String existingImageUrl,
}) async {
  try {
    String imageUrl = existingImageUrl;
    if (newImageFile != null) {
      imageUrl = await _uploadImage(newImageFile);
    }
    // Update only specified fields on the document
    await _productsCollection.doc(productId).update({
      'name': name, 'price': price, 'imageUrl': imageUrl,
    });
    await fetchProducts();
  } catch (e) { return e.toString(); }
}

DELETE Operation

// Removing a document from Firestore
Future<String?> deleteProduct(String id) async {
  try {
    await _productsCollection
      .doc(id).delete();

    _products.removeWhere((p) => p.id == id);
    notifyListeners();
    return null;
  } catch (e) { return e.toString(); }
}
// Always confirm before delete!
showDialog<bool>(
  builder: (ctx) => AlertDialog(
    title: Text('Delete Product?'),
    actions: [
      TextButton(...), // Cancel
      TextButton(...), // Confirm
    ],
  ),
);

⚠️ Never delete without user confirmation — always show a dialog for destructive actions.

Setting Up Routes

// main.dart — 3-step registration pattern
// STEP 1 — Define route name inside the screen class
class ProductDetailScreen extends StatelessWidget {
  static const String routName = '/ProductDetailScreen';
}

// STEP 2 — Register in MaterialApp
MaterialApp(
  routes: {
    LoginScreen.routName: (ctx) => LoginScreen(),
    ProductDetailScreen.routName: (ctx) => ProductDetailScreen(),
  },
  // STEP 3 — Handle routes that require arguments
  onGenerateRoute: (settings) {
    if (settings.name == ProductDetailScreen.routName) {
      final product = settings.arguments as ProductModel;
      return MaterialPageRoute(
        builder: (ctx) => ProductDetailScreen(product: product));
    }
    return null;
  },
);

Passing Data Between Screens

// Two approaches — constructor vs. named route arguments

Method 1 — Constructor

// Navigate
Navigator.push(context,
  MaterialPageRoute(
    builder: (_) => ProductDetailScreen(
      product: product)));

// Receive
class ProductDetailScreen {
  final ProductModel product;
  const ProductDetailScreen({required this.product});
}

Method 2 — Named + Arguments

// Navigate
Navigator.pushNamed(
  context,
  ProductDetailScreen.routName,
  arguments: product,
);

// Receive
final product = ModalRoute.of(context)!
  .settings.arguments as ProductModel;
Slide 11 / 15

Practical — Products List

// ProductsScreen.dart — Consumer pattern
class ProductsScreen extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: Consumer<ProductProvider>(
        builder: (ctx, provider, _) {
          if (provider.isLoading) return CircularProgressIndicator();
          return ListView.builder(
            itemCount: provider.products.length,
            itemBuilder: (ctx, i) {
              final product = provider.products[i];
              return ListTile(
                title: Text(product.name),
                onTap: () => Navigator.pushNamed(
                  context, ProductDetailScreen.routName,
                  arguments: product),
              );
            },
          );
        },
      ),
    );
  }
}

Refresh Pattern After Navigation

// Reloading data after push returns
// In ProductsScreen.dart — FAB to add a new product
floatingActionButton: FloatingActionButton(
  onPressed: () async {
    // Await the result from the form screen
    final result = await Navigator.push(
      context,
      MaterialPageRoute(builder: (_) => ProductFormScreen()),
    );
    // Refresh only if product was saved successfully
    if (result == true && mounted) {
      context.read<ProductProvider>().fetchProducts();
    }
  },
  child: Icon(Icons.add),
),

Key Points

  • Use Navigator.pop(context, true) to return a success signal
  • Always check mounted before calling setState or provider methods
  • Refresh data after every Create, Update, and Delete operation

Best Practices Summary

CRUD Best Practices

  • Always show loading indicators
  • Handle errors gracefully with try/catch
  • Refresh data after every mutation
  • Use confirmation dialogs for delete
  • Validate user input before submitting

Navigation Best Practices

  • Use named routes for consistency
  • Pass data via constructor or arguments
  • Always check mounted before setState
  • Use pushReplacement for auth flows
  • Return results from pop when needed
